### COLLECTIVISM ###

#Collectivism 3, bonus on planets without crime.
event = {
	id = est_coll_indv.2
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = est_collectivism_3_picked_flag
	}
	
	immediate = {
		every_country = {
			limit = {
				has_swapped_tradition = tr_est_collectivism_3
			}
			every_owned_planet = {
				if = {
					limit = {
						planet_crime < 1
					}
					if = {
						limit = {
							Not = {
								has_modifier = mod_est_collectivism_3
							}
						}
						add_modifier = {
							modifier = "mod_est_collectivism_3"
							days = -1
						}	
					}
				}		
				else = {
					if = {
						limit = {
							has_modifier = mod_est_collectivism_3
						}
						remove_modifier = "mod_est_collectivism_3"
					}
				}			
			}
		}
	}
}

#Individualism adopt, influence on leader level up.
country_event = {
	id = est_coll_indv.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_swapped_tradition = tr_est_individualism_adopt
		from = {
			Or = {
				has_skill > 2
				Not = {
					root = {
						has_swapped_tradition = tr_est_individualism_finish
					}
				}
			}
		}
	}
	
	immediate = {
		add_resource = { influence = 10 }
	}
}

#Individualism 1 and 2, bonus level to Admirals and Scientist leaders.
country_event = {
	id = est_coll_indv.4
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_swapped_tradition = tr_est_individualism_adopt
		Or = {
			And = {
				Or = {
					has_swapped_tradition = tr_est_individualism_1
					has_swapped_tradition = tr_est_individualism_1_AI
				}
				from = { leader_class = scientist }
			}
			And = {
				Or = {
					has_swapped_tradition = tr_est_individualism_2
					has_swapped_tradition = tr_est_individualism_2_AI
				}
				from = { leader_class = admiral }
			}
		}
	}
	
	immediate = {
		every_owned_leader = {
			limit = {
				Or = {
					And = {
						Or = {
							owner = {
								has_swapped_tradition = tr_est_individualism_1
								has_swapped_tradition = tr_est_individualism_1_AI
							}
						}
						leader_class = scientist
					}
					And = {
						Or = {
							owner = {
								has_swapped_tradition = tr_est_individualism_2
								has_swapped_tradition = tr_est_individualism_2_AI
							}
						}
						leader_class = admiral
					}
				}
				has_skill < 2
			}
			set_timed_leader_flag = { 
				flag = flag_est_new_leader 
				days = 1
			}
			if = {
				limit = {
					owner = {
						has_non_swapped_tradition = tr_est_academy_2
					}
				}
				if = {
					limit = {
						owner = {
							has_non_swapped_tradition = tr_est_academy_3
						}
					}
					if = {
						limit = {
							owner = {
								has_swapped_tradition = tr_est_individualism_4
							}
						}
						add_experience = 328
					}
					else = {
						add_experience = 380
					}
				}
				else = {
					if = {
						limit = {
							owner = {
								has_swapped_tradition = tr_est_individualism_4
							}
						}
					}
					else = {
						add_experience = 475						
					}
				}
				add_experience = 396
			}
			else = {
				if = {
					limit = {
						owner = {
							has_swapped_tradition = tr_est_individualism_4
						}
					}
					add_experience = 167
				}
				else = {
					add_experience = 200
				}
			}
		}
	}
}

#Perseverance 1, updates when building built.
planet_event = {
	id = est_coll_indv.9
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_swapped_tradition = tr_est_perseverance_1
			has_swapped_tradition = tr_est_perseverance_IP_1
		}
		from = {
			Or = {
				has_building = building_planetary_shield_generator
				last_building_changed = building_planetary_shield_generator
			}
		}
	}
	
	immediate = {
		planet_event = {
			id = est_coll_indv.10
		}
	}
}

#Perseverance 5, fires when war have finished, fires next event with short delay.
country_event = {
	id = est_coll_indv.11
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = est_perseverance_5_picked_flag
		fromfrom = {
			any_war_participant = {
				Or = {
					has_swapped_tradition = tr_est_perseverance_5
					has_swapped_tradition = tr_est_perseverance_IP_5
				}
			}
		}
	}
	
	immediate = {
		fromfrom = {
			every_war_participant = {
				limit = {
					Or = {
						has_swapped_tradition = tr_est_perseverance_5
						has_swapped_tradition = tr_est_perseverance_IP_5
					}
				}
				country_event = {
					id = est_coll_indv.12
					days = 1
				}
			}
		}
	}
}

	}
